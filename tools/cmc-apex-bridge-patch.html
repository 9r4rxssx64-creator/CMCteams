1. Variable globale + helpers (ajouter après les FB_* existants)

// ===== APEX BRIDGE v13.3.70 — Listener planning pending =====
var APEX_BRIDGE_KEY = "ax_cmc_planning_pending";
var _apexBridgeSeen = JSON.parse(localStorage.getItem("cmc_apex_bridge_seen") || "{}");

function _apexBridgePersistSeen(){
  try{ localStorage.setItem("cmc_apex_bridge_seen", JSON.stringify(_apexBridgeSeen)); }catch(_){}
}

function _apexBridgeIsAdmin(){
  return A && A.user && A.user.id === AID; // Kevin only (DESARZENS K = U11804)
}

function _apexBridgeShowToast(plan){
  if(!_apexBridgeIsAdmin()) return;
  if(_apexBridgeSeen[plan.id]) return;
  _apexBridgeSeen[plan.id] = Date.now();
  _apexBridgePersistSeen();
  // Toast doré 1-clic — utilise le pattern toast existant CMCteams
  var div = document.createElement("div");
  div.className = "ax-bridge-toast";
  div.style.cssText = "position:fixed;top:20px;right:20px;z-index:99999;background:#2d2410;border:2px solid #f0c14b;color:#f0c14b;padding:14px 18px;border-radius:8px;box-shadow:0 4px 20px rgba(0,0,0,0.4);font-size:14px;max-width:340px;animation:axBridgeSlide 0.3s ease;cursor:default";
  var safeSrc = (plan.source||"chat").replace(/[^a-z]/gi,"");
  div.innerHTML = "<div style='font-weight:600;margin-bottom:8px'>📥 Apex a envoyé un planning</div>"
    + "<div style='font-size:12px;opacity:0.85;margin-bottom:10px'>"+(plan.original_size||0)+" caractères · source "+esc(safeSrc)+"</div>"
    + "<div style='display:flex;gap:8px'>"
    +   "<button onclick='_apexBridgeImport(\""+esc(plan.id)+"\")' style='flex:1;background:#f0c14b;color:#0d1117;border:none;padding:8px 12px;border-radius:5px;font-weight:600;cursor:pointer'>Importer</button>"
    +   "<button onclick='this.parentElement.parentElement.remove()' style='background:transparent;color:#c9d1d9;border:1px solid #30363d;padding:8px 12px;border-radius:5px;cursor:pointer'>Plus tard</button>"
    + "</div>";
  document.body.appendChild(div);
  // Auto-dismiss 30s
  setTimeout(function(){ if(div.parentNode) div.remove(); }, 30000);
}

function _apexBridgeImport(planId){
  var raw = localStorage.getItem(APEX_BRIDGE_KEY+"/"+planId);
  if(!raw){
    if(typeof toast === "function") toast("⚠ Planning Apex introuvable (peut-être déjà traité)", "warn");
    return;
  }
  try{
    var plan = JSON.parse(raw);
    sv("import"); // Navigue vers la vue Import
    setTimeout(function(){
      var ta = document.querySelector("#importPdfTextarea, textarea[name=importText], textarea.import-textarea");
      if(ta){
        ta.value = plan.raw_text || "";
        ta.scrollIntoView({behavior:"smooth", block:"center"});
        ta.focus();
        if(typeof toast === "function") toast("✅ Planning Apex injecté — clique Importer", "success");
      } else {
        if(typeof toast === "function") toast("⚠ Vue Import non trouvée — copie-colle manuellement", "warn");
      }
    }, 500);
  }catch(e){
    if(typeof toast === "function") toast("⚠ Parse error : "+e.message.slice(0,80), "warn");
  }
}

// Hook dans fbApplyData (déjà appelé par le SSE listener Firebase existant)
function _apexBridgeOnFbData(path, data){
  if(!path || path.indexOf(APEX_BRIDGE_KEY) !== 0) return;
  if(!data) return;
  // path = "ax_cmc_planning_pending/pln_xxx"
  var planId = path.replace(APEX_BRIDGE_KEY+"/", "");
  if(!planId || _apexBridgeSeen[planId]) return;
  // Cache local pour _apexBridgeImport()
  try{ localStorage.setItem(APEX_BRIDGE_KEY+"/"+planId, JSON.stringify(data)); }catch(_){}
  // Si admin connecté → toast doré
  if(_apexBridgeIsAdmin() && data.processed === false){
    _apexBridgeShowToast(data);
  }
}

// Cleanup hebdo des plannings vus > 7j
setInterval(function(){
  var now = Date.now();
  var TTL = 7*24*60*60*1000;
  var changed = false;
  for(var id in _apexBridgeSeen){
    if(now - _apexBridgeSeen[id] > TTL){
      delete _apexBridgeSeen[id];
      try{ localStorage.removeItem(APEX_BRIDGE_KEY+"/"+id); }catch(_){}
      changed = true;
    }
  }
  if(changed) _apexBridgePersistSeen();
}, 60*60*1000);

Notes architecture

  • Cap raw_text 50 KB (sécurité Firebase) déjà géré côté Apex.
  • Cleanup hebdo automatique des plannings vus > 7j.
  • Toast n'apparaît que si A.user.id === AID (Kevin admin only).
  • Anti-double-trigger via _apexBridgeSeen persistant localStorage.
